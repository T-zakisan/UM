% tcolorbox のスタイル定義
\tcbset{
    innerBox/.style={
      colframe  = gray!50, % 枠の色
      colback   = white, % 背景色
      boxrule   = 0.2mm, % 枠線の太さ
      width     = 70mm, % ボックスの幅
      height    = 45mm, % ボックスの高さ
      halign    = center, % 水平方向の中央揃え
      valign    = center, % 垂直方向の中央揃え
      center, % ボックス自体の中央配置
      sharp corners % 角を丸くしない
    },
    innerBoxNone/.style={
      colframe  = white, % 枠の色
      colback   = white, % 背景色
      boxrule   = 0.0mm, % 枠線の太さ
      width     = 70mm, % ボックスの幅
      height    = 45mm, % ボックスの高さ
      halign    = center, % 水平方向の中央揃え
      valign    = top, % 垂直方向の中央揃え
      center, % ボックス自体の中央配置
      sharp corners % 角を丸くしない
    },
    outerBox/.style={
      colframe  = white, % 枠の色
      colback   = white, % 背景色
      boxrule   = 0.0mm, % 枠線の太さ
      width     = 70mm, % ボックスの幅
      height    = 60mm, % ボックスの高さ
      halign    = left, % 水平方向の中央揃え
      valign    = top, % 垂直方向の中央揃え
      center, % ボックス自体の中央配置
      sharp corners % 角を丸くしない
  },
  withImg/.style={
    colframe  = black, % 枠の色
    colback   = red, % 背景色
    boxrule   = 0.0mm, % 枠線の太さ
    width     = \textwidth, % ボックスの幅
    halign    = left, % 水平方向の中央揃え
    valign    = top, % 垂直方向の中央揃え
  }
}
