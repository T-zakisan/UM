% tcolorbox のスタイル定義
\tcbset{
  innerBox/.style={
    colframe  = gray!50, % 枠の色
    colback   = white, % 背景色
    boxrule   = 0.2mm, % 枠線の太さ
    width     = 70mm, % ボックスの幅
    height    = 45mm, % ボックスの高さ
    halign    = center, % 水平方向の中央揃え
    valign    = center, % 垂直方向の中央揃え
    center, % ボックス自体の中央配置
    sharp corners % 角を丸くしない
  },
  innerBoxNone/.style={
    colframe  = white, % 枠の色
    colback   = white, % 背景色
    boxrule   = 0.0mm, % 枠線の太さ
    width     = 70mm, % ボックスの幅
    height    = 45mm, % ボックスの高さ
    halign    = center, % 水平方向の中央揃え
    valign    = top, % 垂直方向の中央揃え
    center, % ボックス自体の中央配置
    sharp corners % 角を丸くしない
  },
  outerBox/.style={
    colframe  = white, % 枠の色
    colback   = white, % 背景色
    boxrule   = 0.0mm, % 枠線の太さ
    width     = 70mm, % ボックスの幅
    height    = 60mm, % ボックスの高さ
    halign    = left, % 水平方向の中央揃え
    valign    = top, % 垂直方向の中央揃え
    center, % ボックス自体の中央配置
    sharp corners % 角を丸くしない
  },
  withImg/.style={
    colframe  = white, % 枠の色
    colback   = white, % 背景色
    boxrule   = 0.0mm, % 枠線の太さ
    width     = \textwidth, % ボックスの幅
    halign    = left, % 水平方向の左寄せ
    valign    = top, % 垂直方向の中央揃え
    top       = 0.0mm,
    % bottom    = 5.0mm,
    left      = -0.3em,
    right     = 0.0mm,
    sharp corners % 角を丸くしない
  }
}


\newsavebox{\imagebox}


% \回込図 コマンドの定義
\NewDocumentCommand{\回込図}{m m}{%
  % #1: 画像ファイルパス (必須)
  % #2: テキスト内容 (必須)
  \vspace{1.8mm}
  \sbox{\imagebox}{%
    \begin{minipage}[t]{70mm}%
      \includegraphics[width=70mm]{#1}%
    \end{minipage}%
  }%
  \begin{minipage}[t]{\dimexpr\linewidth-75mm-\leftskip-0.3em} % 横幅を調整
    \begin{tcolorbox}[withImg, height=\ht\imagebox] % 画像の高さに合わせる
      #2
    \end{tcolorbox}
  \end{minipage}%
  \hspace{\dimexpr 5mm + 0.3em}%
  \usebox{\imagebox}%
}
