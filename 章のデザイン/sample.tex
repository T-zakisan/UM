\documentclass{article}


\usepackage{tcolorbox} %枠囲みを作成する自由度の高いパッケージ


% sectionの再定義（枠付き）
\makeatletter
\renewcommand{\section}{%
    \@ifstar{\customsectionstar}{\customsection}} % *付きと無しで分岐
\newcommand{\customsection}[1]{%
    \cleardoublepage % 右ページから開始
    \thispagestyle{empty} % このページのページ番号を非表示
    \begin{tcolorbox}[
        enhanced,
        colback=gray!20, % 背景色を薄い灰色に設定
        colframe=gray!20, % 枠線を背景色と同じにして非表示
        borderline horizontal={5.5pt}{0.0pt}{black}, % 上下の装飾線のみ表示
        sharp corners % 角を直角に
    ]%
    \LARGE\bfseries\gtfamily\centering \thesection 章 \hspace{1ex} #1
    \end{tcolorbox}%
    \clearpage % 新ページからスタート
}
\newcommand{\customsectionstar}[1]{%
    \cleardoublepage % 右ページから開始
    \thispagestyle{empty} % このページのページ番号を非表示
    \begin{tcolorbox}[
        enhanced,
        colback=gray!20, % 背景色を薄い灰色に設定
        colframe=gray!20, % 枠線を背景色と同じにして非表示
        borderline horizontal={5.5pt}{0.0pt}{black}, % 上下の装飾線のみ表示
        sharp corners % 角を直角に
    ]%
    \LARGE\bfseries\gtfamily\centering #1
    \end{tcolorbox}%
    \clearpage % 新ページからスタート
}
\makeatother
