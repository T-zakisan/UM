% sectionB: 別ページ、章番号なし、ヘッダー・フッターなし、目次にページ数なし
\newcommand{\sectionB}[1]{
  \cleardoublepage % 改ページ
  \newgeometry{top=80truemm, bottom=18truemm, left=23truemm, right=23truemm}
  \thispagestyle{empty} % ページスタイルをemptyに設定
  \markboth{\protect\textcolor{white}{#1}}{} % ここが重要：markbothはタイトルのみ
  \begin{tcolorbox}[colback=gray!20, colframe=black, sharp corners,
                      top=5mm, left=0mm, bottom=5mm, right=0mm,
                      boxrule=0pt, toprule=1.5pt, bottomrule=1.5pt,
                      width=\linewidth ] % 装飾付きボックス
      \fontsize{34pt}{34pt}\selectfont\bfseries\gtfamily\centering #1
  \end{tcolorbox}
\addcontentsline{toc}{section}{#1} % 目次への追加（ページ数なし）
  \newpage
\setcounter{section}{-1} % ページ番号リセット
  \setcounter{page}{1} % ページ番号リセット
  \restoregeometry
}

% sectionC: 別ページ、章番号なし、ヘッダー・フッターあり、目次にページ数なし
\newcommand{\sectionC}[1]{
  \cleardoublepage % 改ページ
  \newgeometry{top=80truemm, bottom=18truemm, left=23truemm, right=23truemm}
  \thispagestyle{empty} % ページスタイルをemptyに設定
  \setcounter{page}{0} % ページ番号リセット
  \markboth{\protect\textcolor{white}{#1}}{} % 白いヘッダー
  \begin{tcolorbox}[colback=gray!20, colframe=black, sharp corners,
                      top=5mm, left=0mm, bottom=5mm, right=0mm,
                      boxrule=0pt, toprule=1.5pt, bottomrule=1.5pt,
                      width=\linewidth ] % 装飾付きボックス
      \fontsize{34pt}{34pt}\selectfont\bfseries\gtfamily\centering #1
  \end{tcolorbox}
  \newpage
\setcounter{section}{0} % ページ番号リセット
  \setcounter{page}{1} % ページ番号リセット
% \addcontentsline{toc}{section}{#1\hfill P=\thesectionpage} % 目次への追加（ページ数なし）
\addcontentsline{toc}{section}{#1}% \hfill\thepage を削除
\phantomsection % hyperrefのためのアンカーを設定
  \restoregeometry
}


% sectionD: 別ページ、章番号あり、ヘッダー・フッターあり、目次にページ数あり
\newcommand{\sectionD}[1]{
  \cleardoublepage % 改ページ
  \newgeometry{top=80truemm, bottom=18truemm, left=23truemm, right=23truemm}
  \thispagestyle{empty} % ページスタイルをemptyに設定
  \markboth{\protect\textcolor{white}{#1}}{} % 白いヘッダー
\refstepcounter{section} % sectionカウンタを進める
  \begin{tcolorbox}[colback=gray!20, colframe=black, sharp corners,
                      top=5mm, left=0mm, bottom=5mm, right=0mm,
                      boxrule=0pt, toprule=1.5pt, bottomrule=1.5pt,
                      width=\linewidth ] % 装飾付きボックス
\fontsize{34pt}{34pt}\selectfont\bfseries\gtfamily\centering \thesection 章 \hspace{1em}#1 % 本文での変更
  \end{tcolorbox}
  \newpage
  \setcounter{page}{1} % ページ番号リセット
% \addcontentsline{toc}{section}{\thesection 章\hspace{1em}#1\hfill P=\thesectionpage} % 目次への追加（ページ数あり）
\addcontentsline{toc}{section}{\thesection 章\hspace{1em}#1} % 目次への追加（ページ数あり）
\phantomsection % hyperrefのためのアンカーを設定
  \restoregeometry
}
